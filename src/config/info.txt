=========== Comandos ===========
iniciar: npm run start:dev 

reinstalar: npm install
instalar algo: npm i -lo_que_vaya_a_instalar

módulos: nest g mo modules/nombre-archivo
controladores: nest g co modules/nombre-archivo --no-spec
servicios: nest g s modules/nombre-archivo --no-spec
